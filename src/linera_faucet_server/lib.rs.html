121/// Persistent mapping of account owners to chain descriptions
122#[derive(Debug, Clone, Serialize, Deserialize, Default)]
123struct FaucetStorage {
124    /// Maps account owners to their corresponding chain descriptions
125    owner_to_chain: HashMap<AccountOwner, ChainDescription>,
126}
127
128impl FaucetStorage {
129    /// Loads the faucet storage from disk, creating a new one if it doesn't exist
130    async fn load(storage_path: &PathBuf) -> Result<Self, io::Error> {
131        match tokio::fs::read(storage_path).await {
132            Ok(data) => serde_json::from_slice(&data).map_err(|e| {
133                io::Error::new(
134                    io::ErrorKind::InvalidData,
135                    format!("Failed to deserialize faucet storage: {}", e),
136                )
137            }),
138            Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(Self::default()),
139            Err(e) => Err(e),
140        }
141    }
142
143    /// Saves the faucet storage to disk
144    async fn save(&self, storage_path: &PathBuf) -> Result<(), io::Error> {
145        let data = serde_json::to_vec_pretty(self).map_err(|e| {
146            io::Error::new(
147                io::ErrorKind::InvalidData,
148                format!("Failed to serialize faucet storage: {}", e),
149            )
150        })?;
151
152        // Create parent directory if it doesn't exist
153        if let Some(parent) = storage_path.parent() {
154            tokio::fs::create_dir_all(parent).await?;
155        }
156
157        tokio::fs::write(storage_path, data).await
158    }
159
160    /// Gets the chain description for an owner if it exists
161    fn get_chain(&self, owner: &AccountOwner) -> Option<&ChainDescription> {
162        self.owner_to_chain.get(owner)
163    }
164
165    /// Stores a new mapping from owner to chain description
166    fn store_chain(&mut self, owner: AccountOwner, description: ChainDescription) {
167        self.owner_to_chain.insert(owner, description);
168    }
169}

208impl MutationRoot {
209    async fn do_claim(&self, owner: AccountOwner) -> Result<ChainDescription, Error> {
210        // Check if this owner already has a chain
211        {
212            let storage = self.faucet_storage.lock().await;
213            if let Some(existing_description) = storage.get_chain(&owner) {
214                return Ok(existing_description.clone());
215            }
216        }
217
218        // Create a oneshot channel to receive the result
219        let (tx, rx) = oneshot::channel();
220
221        // Add request to the queue
222        {
223            let mut requests = self.pending_requests.lock().await;
224            requests.push_back(PendingRequest {
225                owner,
226                responder: tx,
227            });
228        }
229
230        // Notify the batch processor that there's a new request
231        self.request_notifier.notify_one();
232
233        // Wait for the result
234        rx.await
235            .map_err(|_| Error::new("Request processing was cancelled"))?
236    }
237}
238/// Multiplies a `u128` with a `u64` and returns the result as a 192-bit number.
239fn multiply(a: u128, b: u64) -> [u64; 3] {
240    let lower = u128::from(u64::MAX);
241    let b = u128::from(b);
242    let mut a1 = (a >> 64) * b;
243    let a0 = (a & lower) * b;
244    a1 += a0 >> 64;
245    [(a1 >> 64) as u64, (a1 & lower) as u64, (a0 & lower) as u64]
246}

248impl<C> BatchProcessor<C>
249where
250    C: ClientContext + 'static,
251{
252    /// Creates a new batch processor
253    fn new(
254        config: BatchProcessorConfig,
255        context: Arc<Mutex<C>>,
256        client: ChainClient<C::Environment>,
257        faucet_storage: Arc<Mutex<FaucetStorage>>,
258        pending_requests: Arc<Mutex<VecDeque<PendingRequest>>>,
259        request_notifier: Arc<Notify>,
260    ) -> Self {
261        Self {
262            config,
263            context,
264            client,
265            faucet_storage,
266            pending_requests,
267            request_notifier,
268        }
269    }
270
271    /// Runs the batch processor loop
272    async fn run(&mut self, cancellation_token: CancellationToken) {
273        loop {
274            tokio::select! {
275                _ = self.request_notifier.notified() => {
276                    if let Err(e) = self.process_batch().await {
277                        tracing::error!("Batch processing error: {}", e);
278                    }
279                }
280                _ = cancellation_token.cancelled() => {
281                    // Process any remaining requests before shutting down
282                    if let Err(e) = self.process_batch().await {
283                        tracing::error!("Final batch processing error: {}", e);
284                    }
285                    break;
286                }
287            }
288        }
289    }
290
291    /// Processes batches until there are no more pending requests in the queue.
292    async fn process_batch(&mut self) -> Result<(), anyhow::Error> {
293        loop {
294            let max_batch_size = self.config.max_batch_size;
295            let mut batch_requests = Vec::new();
296
297            // Collect requests from the queue.
298            {
299                let mut requests = self.pending_requests.lock().await;
300                while batch_requests.len() < max_batch_size && !requests.is_empty() {
301                    if let Some(request) = requests.pop_front() {
302                        batch_requests.push(request);
303                    }
304                }
305            }
306
307            if batch_requests.is_empty() {
308                return Ok(());
309            }
310
311            tracing::info!("Processing batch of {} requests", batch_requests.len());
312
313            if let Err(err) = self.execute_batch(batch_requests).await {
314                tracing::error!("Failed to execute batch: {}", err);
315                return Err(err);
316            }
317        }
318    }
319
320    /// Executes a batch of chain creation requests
321    async fn execute_batch(
322        &mut self,
323        batch_requests: Vec<PendingRequest>,
324    ) -> Result<(), anyhow::Error> {
325        // Pre-validate: check rate limiting and existing chains
326        let mut valid_requests = Vec::new();
327
328        for request in batch_requests {
329            // Check if this owner already has a chain
330            {
331                let storage = self.faucet_storage.lock().await;
332                if let Some(existing_description) = storage.get_chain(&request.owner) {
333                    let _ = request.responder.send(Ok(existing_description.clone()));
334                    continue;
335                }
336            }
337
338            valid_requests.push(request);
339        }
340
341        if valid_requests.is_empty() {
342            return Ok(());
343        }
344
345        // Rate limiting check for the batch
346        if self.config.start_timestamp < self.config.end_timestamp {
347            let local_time = self.client.storage_client().clock().current_time();
348            if local_time < self.config.end_timestamp {
349                let full_duration = self
350                    .config
351                    .end_timestamp
352                    .delta_since(self.config.start_timestamp)
353                    .as_micros();
354                let remaining_duration = self
355                    .config
356                    .end_timestamp
357                    .delta_since(local_time)
358                    .as_micros();
359                let balance = self.client.local_balance().await?;
360
361                let total_amount = self
362                    .config
363                    .amount
364                    .saturating_mul(valid_requests.len() as u128);
365                let Ok(remaining_balance) = balance.try_sub(total_amount) else {
366                    // Not enough balance - reject all requests
367                    for request in valid_requests {
368                        let _ = request
369                            .responder
370                            .send(Err(Error::new("The faucet is empty.")));
371                    }
372                    return Ok(());
373                };
374
375                if multiply(u128::from(self.config.start_balance), remaining_duration)
376                    > multiply(u128::from(remaining_balance), full_duration)
377                {
378                    // Rate limit exceeded - reject all requests
379                    for request in valid_requests {
380                        let _ = request.responder.send(Err(Error::new(
381                            "Not enough unlocked balance; try again later.",
382                        )));
383                    }
384                    return Ok(());
385                }
386            }
387        }
388
389        // Create OpenChain operations for all valid requests
390        let mut operations = Vec::new();
391        for request in &valid_requests {
392            let config = OpenChainConfig {
393                ownership: ChainOwnership::single(request.owner),
394                balance: self.config.amount,
395                application_permissions: ApplicationPermissions::default(),
396            };
397            operations.push(Operation::system(SystemOperation::OpenChain(config)));
398        }
399
400        // Execute all operations in a single block
401        let result = self.client.execute_operations(operations, vec![]).await;
402        self.context
403            .lock()
404            .await
405            .update_wallet(&self.client)
406            .await?;
407
408        let certificate = match result {
409            Err(ChainClientError::LocalNodeError(LocalNodeError::WorkerError(
410                WorkerError::ChainError(chain_err),
411            ))) => {
412                match *chain_err {
413                    ChainError::ExecutionError(exec_err, ChainExecutionContext::Operation(i))
414                        if i > 0
415                            && matches!(
416                                *exec_err,
417                                ExecutionError::BlockTooLarge
418                                    | ExecutionError::FeesExceedFunding { .. }
419                                    | ExecutionError::InsufficientBalance { .. }
420                                    | ExecutionError::MaximumFuelExceeded(_)
421                            ) =>
422                    {
423                        tracing::error!(%exec_err, "Execution of operation {i} failed; reducing batch size");
424                        self.config.max_batch_size = i as usize;
425                        // Put the valid requests back into the queue.
426                        let mut pending_requests = self.pending_requests.lock().await;
427                        for request in valid_requests.into_iter().rev() {
428                            pending_requests.push_front(request);
429                        }
430                        return Ok(()); // Don't return an error, so we retry.
431                    }
432                    chain_err => {
433                        return Err(
434                            ChainClientError::LocalNodeError(LocalNodeError::WorkerError(
435                                WorkerError::ChainError(chain_err.into()),
436                            ))
437                            .into(),
438                        )
439                    }
440                }
441            }
442            Err(err) => return Err(err.into()),
443            Ok(ClientOutcome::Committed(certificate)) => certificate,
444            Ok(ClientOutcome::WaitForTimeout(timeout)) => {
445                let error_msg = format!(
446                    "This faucet is using a multi-owner chain and is not the leader right now. \
447                    Try again at {}",
448                    timeout.timestamp,
449                );
450                for request in valid_requests {
451                    let _ = request.responder.send(Err(Error::new(error_msg.clone())));
452                }
453                return Ok(());
454            }
455        };
456
457        // Parse chain descriptions from the block's blobs
458        let blobs = certificate.block().body.blobs.iter().flatten();
459        let chain_descriptions = blobs
460            .map(|blob| bcs::from_bytes::<ChainDescription>(blob.bytes()))
461            .collect::<Result<Vec<ChainDescription>, _>>()?;
462
463        if chain_descriptions.len() != valid_requests.len() {
464            let error_msg = format!(
465                "Mismatch between operations ({}) and results ({})",
466                valid_requests.len(),
467                chain_descriptions.len()
468            );
469            for request in valid_requests {
470                let _ = request.responder.send(Err(Error::new(error_msg.clone())));
471            }
472            return Err(anyhow::anyhow!(error_msg));
473        }
474
475        // Store results and respond to requests
476        {
477            let mut storage = self.faucet_storage.lock().await;
478            for (request, description) in valid_requests
479                .into_iter()
480                .zip(chain_descriptions.into_iter())
481            {
482                storage.store_chain(request.owner, description.clone());
483                let _ = request.responder.send(Ok(description));
484            }
485
486            if let Err(e) = storage.save(&self.config.storage_path).await {
487                tracing::warn!("Failed to save faucet storage: {}", e);
488            }
489        }
490
491        Ok(())
492    }
493}

569    /// Creates a new instance of the faucet service.
570    pub async fn new(
571        config: FaucetConfig,
572        context: C,
573        storage: <C::Environment as linera_core::Environment>::Storage,
574    ) -> anyhow::Result<Self> {
575        let client = context.make_chain_client(config.chain_id);
576        let context = Arc::new(Mutex::new(context));
577        let start_timestamp = client.storage_client().clock().current_time();
578        client.process_inbox().await?;
579        let start_balance = client.local_balance().await?;
580
581        // Create storage path: use provided path or create temporary directory
582        let (storage_path, temp_dir) = match config.storage_path {
583            Some(path) => (path, None),
584            None => {
585                let temp_dir = tempfile::tempdir()
586                    .context("Failed to create temporary directory for faucet storage")?;
587                let storage_path = temp_dir.path().join("faucet_storage.json");
588                (storage_path, Some(Arc::new(temp_dir)))
589            }
590        };
591
592        // Load the faucet storage
593        let faucet_storage = FaucetStorage::load(&storage_path)
594            .await
595            .context("Failed to load faucet storage")?;
596        let faucet_storage = Arc::new(Mutex::new(faucet_storage));
597
598        // Initialize batching components
599        let pending_requests = Arc::new(Mutex::new(VecDeque::new()));
600        let request_notifier = Arc::new(Notify::new());
601
602        Ok(Self {
603            chain_id: config.chain_id,
604            context,
605            client,
606            genesis_config: config.genesis_config,
607            config: config.chain_listener_config,
608            storage,
609            port: config.port,
610            #[cfg(feature = "metrics")]
611            metrics_port: config.metrics_port,
612            amount: config.amount,
613            end_timestamp: config.end_timestamp,
614            start_timestamp,
615            start_balance,
616            faucet_storage,
617            storage_path,
618            _temp_dir: temp_dir,
619            pending_requests,
620            request_notifier,
621            max_batch_size: config.max_batch_size,
622        })
623    }

629            request_notifier: Arc::clone(&self.request_notifier),

638    #[cfg(feature = "metrics")]

640        SocketAddr::from(([0, 0, 0, 0], self.metrics_port))
641    }

643    /// Runs the faucet.
644    #[tracing::instrument(name = "FaucetService::run", skip_all, fields(port = self.port, chain_id = ?self.chain_id))]
645    pub async fn run(self, cancellation_token: CancellationToken) -> anyhow::Result<()> {
646        let port = self.port;
647        let index_handler = axum::routing::get(graphiql).post(Self::index_handler);
648
649        #[cfg(feature = "metrics")]
650        prometheus_server::start_metrics(self.metrics_address(), cancellation_token.clone());
651
652        let app = Router::new()
653            .route("/", index_handler)
654            .route("/ready", axum::routing::get(|| async { "ready!" }))
655            .route_service("/ws", GraphQLSubscription::new(self.schema()))
656            .layer(Extension(self.clone()))
657            .layer(CorsLayer::permissive());
658
659        info!("GraphiQL IDE: http://localhost:{}", port);
660
661        // Start the batch processor
662        let batch_processor_config = BatchProcessorConfig {
663            amount: self.amount,
664            end_timestamp: self.end_timestamp,
665            start_timestamp: self.start_timestamp,
666            start_balance: self.start_balance,
667            storage_path: self.storage_path.clone(),
668            max_batch_size: self.max_batch_size,
669        };
670        let mut batch_processor = BatchProcessor::new(
671            batch_processor_config,
672            Arc::clone(&self.context),
673            self.client.clone(),
674            Arc::clone(&self.faucet_storage),
675            Arc::clone(&self.pending_requests),
676            Arc::clone(&self.request_notifier),
677        );
678
679        let chain_listener = ChainListener::new(
680            self.config,
681            self.context,
682            self.storage,
683            cancellation_token.clone(),
684        )
685        .run();
686        let batch_processor_task = batch_processor.run(cancellation_token.clone());
687        let tcp_listener =
688            tokio::net::TcpListener::bind(SocketAddr::from(([0, 0, 0, 0], port))).await?;
689        let server = axum::serve(tcp_listener, app).into_future();
690        futures::select! {
691            result = Box::pin(chain_listener).fuse() => result?,
692            _ = Box::pin(batch_processor_task).fuse() => {},
693            result = Box::pin(server).fuse() => result?,
694        };
695
696        Ok(())
697    }
