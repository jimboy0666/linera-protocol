227#[derive(Default)]
228struct Batch {
229    key_value_bytes: Vec<(Vec<u8>, Vec<u8>)>,
230}
231
232impl Batch {
233    fn new() -> Self {
234        Self::default()
235    }
236
237    fn put_key_value_bytes(&mut self, key: Vec<u8>, value: Vec<u8>) {
238        self.key_value_bytes.push((key, value));
239    }
240
241    fn put_key_value<T: Serialize>(&mut self, key: Vec<u8>, value: &T) -> Result<(), ViewError> {
242        let bytes = bcs::to_bytes(value)?;
243        self.key_value_bytes.push((key, bytes));
244        Ok(())
245    }
246
247    fn add_blob(&mut self, blob: &Blob) -> Result<(), ViewError> {
248        #[cfg(with_metrics)]
249        metrics::WRITE_BLOB_COUNTER.with_label_values(&[]).inc();
250        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob.id()))?;
251        self.put_key_value_bytes(blob_key.to_vec(), blob.bytes().to_vec());
252        Ok(())
253    }
254
255    fn add_blob_state(&mut self, blob_id: BlobId, blob_state: &BlobState) -> Result<(), ViewError> {
256        let blob_state_key = bcs::to_bytes(&BaseKey::BlobState(blob_id))?;
257        self.put_key_value(blob_state_key.to_vec(), blob_state)?;
258        Ok(())
259    }
260
261    fn add_certificate(
262        &mut self,
263        certificate: &ConfirmedBlockCertificate,
264    ) -> Result<(), ViewError> {
265        #[cfg(with_metrics)]
266        metrics::WRITE_CERTIFICATE_COUNTER
267            .with_label_values(&[])
268            .inc();
269        let hash = certificate.hash();
270        let cert_key = bcs::to_bytes(&BaseKey::Certificate(hash))?;
271        let block_key = bcs::to_bytes(&BaseKey::ConfirmedBlock(hash))?;
272        self.put_key_value(cert_key.to_vec(), &certificate.lite_certificate())?;
273        self.put_key_value(block_key.to_vec(), certificate.value())?;
274        Ok(())
275    }
276
277    fn add_event(&mut self, event_id: EventId, value: Vec<u8>) -> Result<(), ViewError> {
278        #[cfg(with_metrics)]
279        metrics::WRITE_EVENT_COUNTER.with_label_values(&[]).inc();
280        let event_key = bcs::to_bytes(&BaseKey::Event(event_id))?;
281        self.put_key_value_bytes(event_key.to_vec(), value);
282        Ok(())
283    }
284
285    fn add_network_description(
286        &mut self,
287        information: &NetworkDescription,
288    ) -> Result<(), ViewError> {
289        #[cfg(with_metrics)]
290        metrics::WRITE_NETWORK_DESCRIPTION
291            .with_label_values(&[])
292            .inc();
293        let key = bcs::to_bytes(&BaseKey::NetworkDescription)?;
294        self.put_key_value(key, information)?;
295        Ok(())
296    }
297}

401/// An implementation of [`DualStoreRootKeyAssignment`] that stores the
402/// chain states into the first store.
403#[derive(Clone, Copy)]
404pub struct ChainStatesFirstAssignment;
405
406impl DualStoreRootKeyAssignment for ChainStatesFirstAssignment {
407    fn assigned_store(root_key: &[u8]) -> Result<StoreInUse, bcs::Error> {
408        if root_key.is_empty() {
409            return Ok(StoreInUse::Second);
410        }
411        let store = match bcs::from_bytes(root_key)? {
412            BaseKey::ChainState(_) => StoreInUse::First,
413            _ => StoreInUse::Second,
414        };
415        Ok(store)
416    }
417}

419/// A `Clock` implementation using the system clock.
420#[derive(Clone)]
421pub struct WallClock;
422
423#[cfg_attr(not(web), async_trait)]
424#[cfg_attr(web, async_trait(?Send))]
425impl Clock for WallClock {
426    fn current_time(&self) -> Timestamp {
427        Timestamp::now()
428    }
429
430    async fn sleep(&self, delta: TimeDelta) {
431        linera_base::time::timer::sleep(delta.as_duration()).await
432    }
433
434    async fn sleep_until(&self, timestamp: Timestamp) {
435        let delta = timestamp.delta_since(Timestamp::now());
436        if delta > TimeDelta::ZERO {
437            self.sleep(delta).await
438        }
439    }
440}

442#[cfg(with_testing)]
443#[derive(Default)]
444struct TestClockInner {
445    time: Timestamp,
446    sleeps: BTreeMap<Reverse<Timestamp>, Vec<oneshot::Sender<()>>>,
447}
448
449#[cfg(with_testing)]
450impl TestClockInner {
451    fn set(&mut self, time: Timestamp) {
452        self.time = time;
453        let senders = self.sleeps.split_off(&Reverse(time));
454        for sender in senders.into_values().flatten() {
455            let _ = sender.send(());
456        }
457    }
458
459    fn add_sleep(&mut self, delta: TimeDelta) -> Receiver<()> {
460        self.add_sleep_until(self.time.saturating_add(delta))
461    }
462
463    fn add_sleep_until(&mut self, time: Timestamp) -> Receiver<()> {
464        let (sender, receiver) = oneshot::channel();
465        if self.time >= time {
466            let _ = sender.send(());
467        } else {
468            self.sleeps.entry(Reverse(time)).or_default().push(sender);
469        }
470        receiver
471    }
472}

474/// A clock implementation that uses a stored number of microseconds and that can be updated
475/// explicitly. All clones share the same time, and setting it in one clone updates all the others.
476#[cfg(with_testing)]
477#[derive(Clone, Default)]
478pub struct TestClock(Arc<std::sync::Mutex<TestClockInner>>);
479
480#[cfg(with_testing)]
481#[cfg_attr(not(web), async_trait)]
482#[cfg_attr(web, async_trait(?Send))]
483impl Clock for TestClock {
484    fn current_time(&self) -> Timestamp {
485        self.lock().time
486    }
487
488    async fn sleep(&self, delta: TimeDelta) {
489        if delta == TimeDelta::ZERO {
490            return;
491        }
492        let receiver = self.lock().add_sleep(delta);
493        let _ = receiver.await;
494    }
495
496    async fn sleep_until(&self, timestamp: Timestamp) {
497        let receiver = self.lock().add_sleep_until(timestamp);
498        let _ = receiver.await;
499    }
500}
501
502#[cfg(with_testing)]
503impl TestClock {
504    /// Creates a new clock with its time set to 0, i.e. the Unix epoch.
505    pub fn new() -> Self {
506        TestClock(Arc::default())
507    }
508
509    /// Sets the current time.
510    pub fn set(&self, time: Timestamp) {
511        self.lock().set(time);
512    }
513
514    /// Advances the current time by the specified delta.
515    pub fn add(&self, delta: TimeDelta) {
516        let mut guard = self.lock();
517        let time = guard.time.saturating_add(delta);
518        guard.set(time);
519    }
520
521    /// Returns the current time according to the test clock.
522    pub fn current_time(&self) -> Timestamp {
523        self.lock().time
524    }
525
526    fn lock(&self) -> std::sync::MutexGuard<TestClockInner> {
527        self.0.lock().expect("poisoned TestClock mutex")
528    }
529}

548    async fn load_chain(
549        &self,
550        chain_id: ChainId,
551    ) -> Result<ChainStateView<Self::Context>, ViewError> {
552        #[cfg(with_metrics)]
553        let _metric = metrics::LOAD_CHAIN_LATENCY.measure_latency();
554        let runtime_context = ChainRuntimeContext {
555            storage: self.clone(),
556            chain_id,
557            execution_runtime_config: self.execution_runtime_config,
558            user_contracts: self.user_contracts.clone(),
559            user_services: self.user_services.clone(),
560        };
561        let root_key = bcs::to_bytes(&BaseKey::ChainState(chain_id))?;
562        let store = self.database.open_exclusive(&root_key)?;
563        let context = ViewContext::create_root_context(store, runtime_context).await?;
564        ChainStateView::load(context).await
565    }

567    async fn contains_blob(&self, blob_id: BlobId) -> Result<bool, ViewError> {
568        let store = self.database.open_shared(&[])?;
569        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob_id))?;
570        let test = store.contains_key(&blob_key).await?;
571        #[cfg(with_metrics)]
572        metrics::CONTAINS_BLOB_COUNTER.with_label_values(&[]).inc();
573        Ok(test)
574    }

576    async fn missing_blobs(&self, blob_ids: &[BlobId]) -> Result<Vec<BlobId>, ViewError> {
577        let store = self.database.open_shared(&[])?;
578        let mut keys = Vec::new();
579        for blob_id in blob_ids {
580            let key = bcs::to_bytes(&BaseKey::Blob(*blob_id))?;
581            keys.push(key);
582        }
583        let results = store.contains_keys(keys).await?;
584        let mut missing_blobs = Vec::new();
585        for (blob_id, result) in blob_ids.iter().zip(results) {
586            if !result {
587                missing_blobs.push(*blob_id);
588            }
589        }
590        #[cfg(with_metrics)]
591        metrics::CONTAINS_BLOBS_COUNTER.with_label_values(&[]).inc();
592        Ok(missing_blobs)
593    }

620    async fn read_blob(&self, blob_id: BlobId) -> Result<Option<Blob>, ViewError> {
621        let store = self.database.open_shared(&[])?;
622        let blob_key = bcs::to_bytes(&BaseKey::Blob(blob_id))?;
623        let maybe_blob_bytes = store.read_value_bytes(&blob_key).await?;
624        #[cfg(with_metrics)]
625        metrics::READ_BLOB_COUNTER.with_label_values(&[]).inc();
626        Ok(maybe_blob_bytes.map(|blob_bytes| Blob::new_with_id_unchecked(blob_id, blob_bytes)))
627    }
628
629    async fn read_blobs(&self, blob_ids: &[BlobId]) -> Result<Vec<Option<Blob>>, ViewError> {
630        if blob_ids.is_empty() {
631            return Ok(Vec::new());
632        }
633        let blob_keys = blob_ids
634            .iter()
635            .map(|blob_id| bcs::to_bytes(&BaseKey::Blob(*blob_id)))
636            .collect::<Result<Vec<_>, _>>()?;
637        let store = self.database.open_shared(&[])?;
638        let maybe_blob_bytes = store.read_multi_values_bytes(blob_keys).await?;
639        #[cfg(with_metrics)]
640        metrics::READ_BLOB_COUNTER
641            .with_label_values(&[])

693    async fn maybe_write_blob_states(
694        &self,
695        blob_ids: &[BlobId],
696        blob_state: BlobState,
697    ) -> Result<(), ViewError> {
698        if blob_ids.is_empty() {
699            return Ok(());
700        }
701        let blob_state_keys = blob_ids
702            .iter()
703            .map(|blob_id| bcs::to_bytes(&BaseKey::BlobState(*blob_id)))
704            .collect::<Result<_, _>>()?;
705        let store = self.database.open_shared(&[])?;
706        let maybe_blob_states = store
707            .read_multi_values::<BlobState>(blob_state_keys)
708            .await?;
709        let mut batch = Batch::new();
710        for (maybe_blob_state, blob_id) in maybe_blob_states.iter().zip(blob_ids) {
711            match maybe_blob_state {
712                None => {
713                    batch.add_blob_state(*blob_id, &blob_state)?;
714                }
715                Some(state) => {
716                    if state.epoch < blob_state.epoch {
717                        batch.add_blob_state(*blob_id, &blob_state)?;
718                    }
719                }
720            }
721        }
722        // We tolerate race conditions because two active chains are likely to
723        // be both from the latest epoch, and otherwise failing to pick the
724        // more recent blob state has limited impact.
725        self.write_batch(batch).await?;
726        Ok(())
727    }

729    async fn maybe_write_blobs(&self, blobs: &[Blob]) -> Result<Vec<bool>, ViewError> {
730        if blobs.is_empty() {
731            return Ok(Vec::new());
732        }
733        let blob_state_keys = blobs
734            .iter()
735            .map(|blob| bcs::to_bytes(&BaseKey::BlobState(blob.id())))
736            .collect::<Result<_, _>>()?;
737        let store = self.database.open_shared(&[])?;
738        let blob_states = store.contains_keys(blob_state_keys).await?;
739        let mut batch = Batch::new();
740        for (blob, has_state) in blobs.iter().zip(&blob_states) {
741            if *has_state {
742                batch.add_blob(blob)?;
743            }
744        }
745        self.write_batch(batch).await?;
746        Ok(blob_states)
747    }

801    async fn read_certificates<I: IntoIterator<Item = CryptoHash> + Send>(
802        &self,
803        hashes: I,
804    ) -> Result<Vec<Option<ConfirmedBlockCertificate>>, ViewError> {
805        let hashes = hashes.into_iter().collect::<Vec<_>>();
806        if hashes.is_empty() {
807            return Ok(Vec::new());
808        }
809        let keys = Self::get_keys_for_certificates(&hashes)?;
810        let store = self.database.open_shared(&[])?;
811        let values = store.read_multi_values_bytes(keys).await;
812        if values.is_ok() {
813            #[cfg(with_metrics)]
814            metrics::READ_CERTIFICATES_COUNTER
815                .with_label_values(&[])
816                .inc_by(hashes.len() as u64);
817        }
818        let values = values?;
819        let mut certificates = Vec::new();
820        for (pair, hash) in values.chunks_exact(2).zip(hashes) {
821            let certificate = Self::deserialize_certificate(pair, hash)?;
822            certificates.push(certificate);
823        }
824        Ok(certificates)
825    }

845    async fn read_events_from_index(
846        &self,
847        chain_id: &ChainId,
848        stream_id: &StreamId,
849        start_index: u32,
850    ) -> Result<Vec<IndexAndEvent>, ViewError> {
851        let mut prefix = vec![INDEX_EVENT_ID];
852        prefix.extend(bcs::to_bytes(chain_id).unwrap());
853        prefix.extend(bcs::to_bytes(stream_id).unwrap());
854        let mut keys = Vec::new();
855        let mut indices = Vec::new();
856        let store = self.database.open_shared(&[])?;
857        for short_key in store.find_keys_by_prefix(&prefix).await? {
858            let index = bcs::from_bytes::<u32>(&short_key)?;
859            if index >= start_index {
860                let mut key = prefix.clone();
861                key.extend(short_key);
862                keys.push(key);
863                indices.push(index);
864            }
865        }
866        let values = store.read_multi_values_bytes(keys).await?;
867        let mut returned_values = Vec::new();
868        for (index, value) in indices.into_iter().zip(values) {
869            let event = value.unwrap();
870            returned_values.push(IndexAndEvent { index, event });
871        }
872        Ok(returned_values)
873    }

921impl<Database, C> DbStorage<Database, C>
922where
923    Database: KeyValueDatabase + Clone + Send + Sync + 'static,
924    Database::Store: KeyValueStore + Clone + Send + Sync + 'static,
925    C: Clock,
926    Database::Error: Send + Sync,
927{
928    fn get_keys_for_certificates(hashes: &[CryptoHash]) -> Result<Vec<Vec<u8>>, ViewError> {
929        Ok(hashes
930            .iter()
931            .flat_map(|hash| {
932                let cert_key = bcs::to_bytes(&BaseKey::Certificate(*hash));
933                let block_key = bcs::to_bytes(&BaseKey::ConfirmedBlock(*hash));
934                vec![cert_key, block_key]
935            })
936            .collect::<Result<_, _>>()?)
937    }
938
939    fn deserialize_certificate(
940        pair: &[Option<Vec<u8>>],
941        hash: CryptoHash,
942    ) -> Result<Option<ConfirmedBlockCertificate>, ViewError> {
943        let Some(cert_bytes) = pair[0].as_ref() else {
944            return Ok(None);
945        };
946        let Some(value_bytes) = pair[1].as_ref() else {
947            return Ok(None);
948        };
949        let cert = bcs::from_bytes::<LiteCertificate>(cert_bytes)?;
950        let value = bcs::from_bytes::<ConfirmedBlock>(value_bytes)?;
951        assert_eq!(value.hash(), hash);
952        let certificate = cert
953            .with_value(value)
954            .ok_or(ViewError::InconsistentEntries)?;
955        Ok(Some(certificate))
956    }
957
958    async fn write_entry(
959        store: &Database::Store,
960        key: Vec<u8>,
961        bytes: Vec<u8>,
962    ) -> Result<(), ViewError> {
963        let mut batch = linera_views::batch::Batch::new();
964        batch.put_key_value_bytes(key, bytes);
965        store.write_batch(batch).await?;
966        Ok(())
967    }
968
969    async fn write_batch(&self, batch: Batch) -> Result<(), ViewError> {
970        if batch.key_value_bytes.is_empty() {
971            return Ok(());
972        }
973        let mut futures = Vec::new();
974        for (key, bytes) in batch.key_value_bytes.into_iter() {
975            let store = self.database.open_shared(&[])?;
976            futures.push(async move { Self::write_entry(&store, key, bytes).await });
977        }
978        futures::future::try_join_all(futures).await?;
979        Ok(())
980    }
981}
