63    /// Creates a new [`ChainWorkerState`] using the provided `storage` client.

66        config: ChainWorkerConfig,
67        storage: StorageClient,
68        block_values: Arc<ValueCache<CryptoHash, Hashed<Block>>>,
69        execution_state_cache: Arc<ValueCache<CryptoHash, ExecutionStateView<InactiveContext>>>,

71        delivery_notifier: DeliveryNotifier,
72        chain_id: ChainId,
73        service_runtime_endpoint: Option<ServiceRuntimeEndpoint>,

75        let chain = storage.load_chain(chain_id).await?;

78            config,
79            storage,
80            chain,

82            service_runtime_endpoint,
83            block_values,
84            execution_state_cache,
85            tracked_chains,
86            delivery_notifier,
87            knows_chain_is_active: false,
88        })
89    }

96    /// Handles a request and applies it to the chain state.
97    #[instrument(skip_all)]
98    pub async fn handle_request(&mut self, request: ChainWorkerRequest<StorageClient::Context>) {
99        tracing::trace!("Handling chain worker request: {request:?}");
100        // TODO(#2237): Spawn concurrent tasks for read-only operations
101        let responded = match request {
102            #[cfg(with_testing)]
103            ChainWorkerRequest::ReadCertificate { height, callback } => callback
104                .send(self.guard().await.read_certificate(height).await)
105                .is_ok(),
106            ChainWorkerRequest::GetChainStateView { callback } => {
107                callback.send(self.chain_state_view().await).is_ok()
108            }
109            ChainWorkerRequest::QueryApplication { query, callback } => callback
110                .send(self.guard().await.query_application(query).await)
111                .is_ok(),
112            ChainWorkerRequest::DescribeApplication {
113                application_id,
114                callback,
115            } => callback
116                .send(
117                    self.guard()
118                        .await
119                        .describe_application(application_id)
120                        .await,
121                )
122                .is_ok(),
123            ChainWorkerRequest::StageBlockExecution {
124                block,
125                round,
126                published_blobs,
127                callback,
128            } => callback
129                .send(
130                    self.guard()
131                        .await
132                        .stage_block_execution(block, round, &published_blobs)
133                        .await,
134                )
135                .is_ok(),
136            ChainWorkerRequest::ProcessTimeout {
137                certificate,
138                callback,
139            } => callback
140                .send(self.guard().await.process_timeout(certificate).await)
141                .is_ok(),
142            ChainWorkerRequest::HandleBlockProposal { proposal, callback } => callback
143                .send(self.guard().await.handle_block_proposal(proposal).await)
144                .is_ok(),
145            ChainWorkerRequest::ProcessValidatedBlock {
146                certificate,
147                callback,
148            } => callback
149                .send(
150                    self.guard()
151                        .await
152                        .process_validated_block(certificate)
153                        .await,
154                )
155                .is_ok(),
156            ChainWorkerRequest::ProcessConfirmedBlock {
157                certificate,
158                notify_when_messages_are_delivered,
159                callback,
160            } => callback
161                .send(
162                    self.guard()
163                        .await
164                        .process_confirmed_block(certificate, notify_when_messages_are_delivered)
165                        .await,
166                )
167                .is_ok(),
168            ChainWorkerRequest::ProcessCrossChainUpdate {
169                origin,
170                bundles,
171                callback,
172            } => callback
173                .send(
174                    self.guard()
175                        .await
176                        .process_cross_chain_update(origin, bundles)
177                        .await,
178                )
179                .is_ok(),
180            ChainWorkerRequest::ConfirmUpdatedRecipient {
181                recipient,
182                latest_height,
183                callback,
184            } => callback
185                .send(
186                    self.guard()
187                        .await
188                        .confirm_updated_recipient(recipient, latest_height)
189                        .await,
190                )
191                .is_ok(),
192            ChainWorkerRequest::HandleChainInfoQuery { query, callback } => callback
193                .send(self.handle_chain_info_query(query).await)
194                .is_ok(),
195            ChainWorkerRequest::DownloadPendingBlob { blob_id, callback } => callback
196                .send(self.download_pending_blob(blob_id).await)
197                .is_ok(),
198            ChainWorkerRequest::HandlePendingBlob { blob, callback } => callback
199                .send(self.guard().await.handle_pending_blob(blob).await)
200                .is_ok(),
201            ChainWorkerRequest::UpdateReceivedCertificateTrackers {
202                new_trackers,
203                callback,
204            } => callback
205                .send(
206                    self.update_received_certificate_trackers(new_trackers)
207                        .await,
208                )
209                .is_ok(),
210        };
211
212        if !responded {
213            warn!("Callback for `ChainWorkerActor` was dropped before a response was sent");
214        }
215    }

217    /// Returns a read-only view of the [`ChainStateView`].
218    ///
219    /// The returned view holds a lock on the chain state, which prevents the worker from changing
220    /// it.
221    pub(super) async fn chain_state_view(
222        &mut self,
223    ) -> Result<OwnedRwLockReadGuard<ChainStateView<StorageClient::Context>>, WorkerError> {
224        if self.shared_chain_view.is_none() {
225            self.shared_chain_view = Some(Arc::new(RwLock::new(self.chain.clone_unchecked())));
226        }
227
228        Ok(self
229            .shared_chain_view
230            .as_ref()
231            .expect("`shared_chain_view` should be initialized above")
232            .clone()
233            .read_owned()
234            .await)
235    }

237    /// Clears the shared chain view, and acquires and drops its write lock.
238    ///
239    /// This is the only place a write lock is acquired, and read locks are acquired in
240    /// the `chain_state_view` method, which has a `&mut self` receiver like this one.
241    /// That means that when this function returns, no readers will be waiting to acquire
242    /// the lock and it is safe to write the chain state to storage without any readers
243    /// having a stale view of it.
244    pub(super) async fn clear_shared_chain_view(&mut self) {
245        if let Some(shared_chain_view) = self.shared_chain_view.take() {
246            let _: RwLockWriteGuard<_> = shared_chain_view.write().await;
247        }
248    }

250    /// Handles a [`ChainInfoQuery`], potentially voting on the next block.
251    #[tracing::instrument(level = "debug", skip(self))]
252    pub(super) async fn handle_chain_info_query(
253        &mut self,
254        query: ChainInfoQuery,
255    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
256        if let Some((height, round)) = query.request_leader_timeout {
257            ChainWorkerGuard::new(&mut *self)
258                .await
259                .vote_for_leader_timeout(height, round)
260                .await?;
261        }
262        if query.request_fallback {
263            ChainWorkerGuard::new(&mut *self)
264                .await
265                .vote_for_fallback()
266                .await?;
267        }
268        let response = self
269            .guard()
270            .await
271            .prepare_chain_info_response(query)
272            .await?;
273        // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
274        let actions = self.create_network_actions().await?;
275        Ok((response, actions))
276    }

278    /// Returns the requested blob, if it belongs to the current locking block or pending proposal.
279    pub(super) async fn download_pending_blob(&self, blob_id: BlobId) -> Result<Blob, WorkerError> {
280        if let Some(blob) = self.chain.manager.pending_blob(&blob_id).await? {
281            return Ok(blob);
282        }
283        let blob = self.storage.read_blob(blob_id).await?;
284        blob.ok_or(WorkerError::BlobsNotFound(vec![blob_id]))
285    }

287    /// Reads the blobs from the chain manager or from storage. Returns an error if any are
288    /// missing.
289    async fn get_required_blobs(
290        &self,
291        required_blob_ids: impl IntoIterator<Item = BlobId>,
292        created_blobs: &BTreeMap<BlobId, Blob>,
293    ) -> Result<BTreeMap<BlobId, Blob>, WorkerError> {
294        let maybe_blobs = self
295            .maybe_get_required_blobs(required_blob_ids, Some(created_blobs))
296            .await?;
297        let not_found_blob_ids = missing_blob_ids(&maybe_blobs);
298        ensure!(
299            not_found_blob_ids.is_empty(),
300            WorkerError::BlobsNotFound(not_found_blob_ids)
301        );
302        Ok(maybe_blobs
303            .into_iter()
304            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
305            .collect())
306    }

308    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
309    async fn maybe_get_required_blobs(
310        &self,
311        blob_ids: impl IntoIterator<Item = BlobId>,
312        created_blobs: Option<&BTreeMap<BlobId, Blob>>,
313    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
314        let mut maybe_blobs = BTreeMap::from_iter(blob_ids.into_iter().zip(iter::repeat(None)));
315
316        for (blob_id, maybe_blob) in &mut maybe_blobs {
317            if let Some(blob) = created_blobs.and_then(|blob_map| blob_map.get(blob_id)) {
318                *maybe_blob = Some(blob.clone());
319            } else if let Some(blob) = self.chain.manager.pending_blob(blob_id).await? {
320                *maybe_blob = Some(blob);
321            } else if let Some(blob) = self.chain.pending_validated_blobs.get(blob_id).await? {
322                *maybe_blob = Some(blob);
323            } else {
324                for (_, pending_blobs) in self
325                    .chain
326                    .pending_proposed_blobs
327                    .try_load_all_entries()
328                    .await?
329                {
330                    if let Some(blob) = pending_blobs.get(blob_id).await? {
331                        *maybe_blob = Some(blob);
332                        break;
333                    }
334                }
335            }
336        }
337        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
338        let blobs_from_storage = self.storage.read_blobs(&missing_blob_ids).await?;
339        for (blob_id, maybe_blob) in missing_blob_ids.into_iter().zip(blobs_from_storage) {
340            maybe_blobs.insert(blob_id, maybe_blob);
341        }
342        Ok(maybe_blobs)
343    }

345    /// Adds any newly created chains to the set of `tracked_chains`, if the parent chain is
346    /// also tracked.
347    ///
348    /// Chains that are not tracked are usually processed only because they sent some message
349    /// to one of the tracked chains. In most use cases, their children won't be of interest.
350    fn track_newly_created_chains(
351        &self,
352        proposed_block: &ProposedBlock,
353        outcome: &BlockExecutionOutcome,
354    ) {
355        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
356            if !tracked_chains
357                .read()
358                .expect("Panics should not happen while holding a lock to `tracked_chains`")
359                .contains(&proposed_block.chain_id)
360            {
361                return; // The parent chain is not tracked; don't track the child.
362            }
363            let new_chain_ids = outcome
364                .created_blobs_ids()
365                .into_iter()
366                .filter(|blob_id| blob_id.blob_type == BlobType::ChainDescription)
367                .map(|blob_id| ChainId(blob_id.hash));
368
369            tracked_chains
370                .write()
371                .expect("Panics should not happen while holding a lock to `tracked_chains`")
372                .extend(new_chain_ids);
373        }
374    }

376    /// Loads pending cross-chain requests.
377    async fn create_network_actions(&self) -> Result<NetworkActions, WorkerError> {
378        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
379        let mut targets = self.chain.nonempty_outbox_chain_ids();
380        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
381            let tracked_chains = tracked_chains
382                .read()
383                .expect("Panics should not happen while holding a lock to `tracked_chains`");
384            targets.retain(|target| tracked_chains.contains(target));
385        }
386        let outboxes = self.chain.load_outboxes(&targets).await?;
387        for (target, outbox) in targets.into_iter().zip(outboxes) {
388            let heights = outbox.queue.elements().await?;
389            heights_by_recipient.insert(target, heights);
390        }
391        self.create_cross_chain_requests(heights_by_recipient).await
392    }

394    async fn create_cross_chain_requests(
395        &self,
396        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
397    ) -> Result<NetworkActions, WorkerError> {
398        // Load all the certificates we will need, regardless of the medium.
399        let heights = BTreeSet::from_iter(heights_by_recipient.values().flatten().copied());
400        let next_block_height = self.chain.tip_state.get().next_block_height;
401        let log_heights = heights
402            .range(..next_block_height)
403            .copied()
404            .map(usize::try_from)
405            .collect::<Result<Vec<_>, _>>()?;
406        let mut hashes = self
407            .chain
408            .confirmed_log
409            .multi_get(log_heights)
410            .await?
411            .into_iter()
412            .zip(&heights)
413            .map(|(maybe_hash, height)| {
414                maybe_hash.ok_or_else(|| WorkerError::ConfirmedLogEntryNotFound {
415                    height: *height,
416                    chain_id: self.chain_id(),
417                })
418            })
419            .collect::<Result<Vec<_>, _>>()?;
420        for height in heights.range(next_block_height..) {
421            hashes.push(
422                self.chain
423                    .preprocessed_blocks
424                    .get(height)
425                    .await?
426                    .ok_or_else(|| WorkerError::PreprocessedBlocksEntryNotFound {
427                        height: *height,
428                        chain_id: self.chain_id(),
429                    })?,
430            );
431        }
432        let certificates = self.storage.read_certificates(hashes.clone()).await?;
433        let certificates = match ResultReadCertificates::new(certificates, hashes) {
434            ResultReadCertificates::Certificates(certificates) => certificates,
435            ResultReadCertificates::InvalidHashes(hashes) => {
436                return Err(WorkerError::ReadCertificatesError(hashes))
437            }
438        };
439        let certificates = heights
440            .into_iter()
441            .zip(certificates)
442            .collect::<HashMap<_, _>>();
443        // For each medium, select the relevant messages.
444        let mut actions = NetworkActions::default();
445        for (recipient, heights) in heights_by_recipient {
446            let mut bundles = Vec::new();
447            for height in heights {
448                let cert = certificates
449                    .get(&height)
450                    .ok_or_else(|| ChainError::InternalError("missing certificates".to_string()))?;
451                bundles.extend(cert.message_bundles_for(recipient));
452            }
453            let request = CrossChainRequest::UpdateRecipient {
454                sender: self.chain.chain_id(),
455                recipient,
456                bundles,
457            };
458            actions.cross_chain_requests.push(request);
459        }
460        Ok(actions)
461    }

463    /// Returns true if there are no more outgoing messages in flight up to the given
464    /// block height.
465    pub async fn all_messages_to_tracked_chains_delivered_up_to(
466        &self,
467        height: BlockHeight,
468    ) -> Result<bool, WorkerError> {
469        if self.chain.all_messages_delivered_up_to(height) {
470            return Ok(true);
471        }
472        let Some(tracked_chains) = self.tracked_chains.as_ref() else {
473            return Ok(false);
474        };
475        let mut targets = self.chain.nonempty_outbox_chain_ids();
476        {
477            let tracked_chains = tracked_chains.read().unwrap();
478            targets.retain(|target| tracked_chains.contains(target));
479        }
480        let outboxes = self.chain.load_outboxes(&targets).await?;
481        for outbox in outboxes {
482            let front = outbox.queue.front();
483            if front.is_some_and(|key| *key <= height) {
484                return Ok(false);
485            }
486        }
487        Ok(true)
488    }

506/// Returns the keys whose value is `None`.
507fn missing_blob_ids(maybe_blobs: &BTreeMap<BlobId, Option<Blob>>) -> Vec<BlobId> {
508    maybe_blobs
509        .iter()
510        .filter(|(_, maybe_blob)| maybe_blob.is_none())
511        .map(|(blob_id, _)| *blob_id)
512        .collect()
513}

515/// Returns an error if the block is not at the expected epoch.
516fn check_block_epoch(
517    chain_epoch: Epoch,
518    block_chain: ChainId,
519    block_epoch: Epoch,
520) -> Result<(), WorkerError> {
521    ensure!(
522        block_epoch == chain_epoch,
523        WorkerError::InvalidEpoch {
524            chain_id: block_chain,
525            epoch: block_epoch,
526            chain_epoch
527        }
528    );
529    Ok(())
