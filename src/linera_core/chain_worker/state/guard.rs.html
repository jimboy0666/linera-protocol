50    /// Creates a new [`ChainWorkerGuard`] instance to change the `state`.
51    pub(super) async fn new(state: &'state mut ChainWorkerState<StorageClient>) -> Self {
52        assert!(

60    /// Processes a leader timeout issued for this multi-owner chain.
61    pub(super) async fn process_timeout(
62        &mut self,
63        certificate: TimeoutCertificate,
64    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
65        // Check that the chain is active and ready for this timeout.
66        // Verify the certificate. Returns a catch-all error to make client code more robust.
67        self.ensure_is_active().await?;
68        let (chain_epoch, committee) = self.0.chain.current_committee()?;
69        ensure!(
70            certificate.inner().epoch() == chain_epoch,
71            WorkerError::InvalidEpoch {
72                chain_id: certificate.inner().chain_id(),
73                chain_epoch,
74                epoch: certificate.inner().epoch()
75            }
76        );
77        certificate.check(committee)?;
78        let mut actions = NetworkActions::default();
79        if self
80            .0
81            .chain
82            .tip_state
83            .get()
84            .already_validated_block(certificate.inner().height())?
85        {
86            return Ok((self.chain_info_response(), actions));
87        }
88        let old_round = self.0.chain.manager.current_round();
89        let timeout_chain_id = certificate.inner().chain_id();
90        let timeout_height = certificate.inner().height();
91        self.0
92            .chain
93            .manager
94            .handle_timeout_certificate(certificate, self.0.storage.clock().current_time());
95        let round = self.0.chain.manager.current_round();
96        if round > old_round {
97            actions.notifications.push(Notification {
98                chain_id: timeout_chain_id,
99                reason: Reason::NewRound {
100                    height: timeout_height,
101                    round,
102                },
103            })
104        }
105        self.save().await?;
106        Ok((self.chain_info_response(), actions))
107    }

109    /// Tries to load all blobs published in this proposal.
110    ///
111    /// If they cannot be found, it creates an entry in `pending_proposed_blobs` so they can be
112    /// submitted one by one.
113    pub(super) async fn load_proposal_blobs(
114        &mut self,
115        proposal: &BlockProposal,
116    ) -> Result<Vec<Blob>, WorkerError> {
117        let owner = proposal.owner();
118        let BlockProposal {
119            content:
120                ProposalContent {
121                    block,
122                    round,
123                    outcome: _,
124                },
125            original_proposal,
126            signature: _,
127        } = proposal;
128
129        let mut maybe_blobs = self
130            .0
131            .maybe_get_required_blobs(proposal.required_blob_ids(), None)
132            .await?;
133        let missing_blob_ids = super::missing_blob_ids(&maybe_blobs);
134        if !missing_blob_ids.is_empty() {
135            let chain = &mut self.0.chain;
136            if chain.ownership().open_multi_leader_rounds {
137                // TODO(#3203): Allow multiple pending proposals on permissionless chains.
138                chain.pending_proposed_blobs.clear();
139            }
140            let validated = matches!(original_proposal, Some(OriginalProposal::Regular { .. }));
141            chain
142                .pending_proposed_blobs
143                .try_load_entry_mut(&owner)
144                .await?
145                .update(*round, validated, maybe_blobs)
146                .await?;
147            self.save().await?;
148            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
149        }
150        let published_blobs = block
151            .published_blob_ids()
152            .iter()
153            .filter_map(|blob_id| maybe_blobs.remove(blob_id).flatten())
154            .collect::<Vec<_>>();
155        Ok(published_blobs)
156    }

158    /// Processes a validated block issued for this multi-owner chain.
159    pub(super) async fn process_validated_block(
160        &mut self,
161        certificate: ValidatedBlockCertificate,
162    ) -> Result<(ChainInfoResponse, NetworkActions, bool), WorkerError> {
163        let block = certificate.block();
164
165        let header = &block.header;
166        let height = header.height;
167        // Check that the chain is active and ready for this validated block.
168        // Verify the certificate. Returns a catch-all error to make client code more robust.
169        self.ensure_is_active().await?;
170        let (epoch, committee) = self.0.chain.current_committee()?;
171        check_block_epoch(epoch, header.chain_id, header.epoch)?;
172        certificate.check(committee)?;
173        let mut actions = NetworkActions::default();
174        let already_committed_block = self
175            .0
176            .chain
177            .tip_state
178            .get()
179            .already_validated_block(height)?;
180        let should_skip_validated_block = || {
181            self.0
182                .chain
183                .manager
184                .check_validated_block(&certificate)
185                .map(|outcome| outcome == manager::Outcome::Skip)
186        };
187        if already_committed_block || should_skip_validated_block()? {
188            // If we just processed the same pending block, return the chain info unchanged.
189            return Ok((self.chain_info_response(), actions, true));
190        }
191
192        self.0
193            .block_values
194            .insert(Cow::Borrowed(certificate.inner().inner()));
195        let required_blob_ids = block.required_blob_ids();
196        let maybe_blobs = self
197            .0
198            .maybe_get_required_blobs(required_blob_ids, Some(&block.created_blobs()))
199            .await?;
200        let missing_blob_ids = super::missing_blob_ids(&maybe_blobs);
201        if !missing_blob_ids.is_empty() {
202            self.0
203                .chain
204                .pending_validated_blobs
205                .update(certificate.round, true, maybe_blobs)
206                .await?;
207            self.save().await?;
208            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
209        }
210        let blobs = maybe_blobs
211            .into_iter()
212            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
213            .collect();
214        let old_round = self.0.chain.manager.current_round();
215        self.0.chain.manager.create_final_vote(
216            certificate,
217            self.0.config.key_pair(),
218            self.0.storage.clock().current_time(),
219            blobs,
220        )?;
221        self.save().await?;
222        let round = self.0.chain.manager.current_round();
223        if round > old_round {
224            actions.notifications.push(Notification {
225                chain_id: self.0.chain_id(),
226                reason: Reason::NewRound { height, round },
227            })
228        }
229        Ok((self.chain_info_response(), actions, false))
230    }

232    /// Processes a confirmed block (aka a commit).
233    pub(super) async fn process_confirmed_block(
234        &mut self,
235        certificate: ConfirmedBlockCertificate,
236        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
237    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
238        let block = certificate.block();
239        let height = block.header.height;
240        let chain_id = block.header.chain_id;
241
242        // Check that the chain is active and ready for this confirmation.
243        let tip = self.0.chain.tip_state.get().clone();
244        if tip.next_block_height > height {
245            // We already processed this block.
246            let actions = self.0.create_network_actions().await?;
247            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
248                .await;
249            return Ok((self.chain_info_response(), actions));
250        }
251
252        // We haven't processed the block - verify the certificate first
253        let epoch = block.header.epoch;
254        // Get the committee for the block's epoch from storage.
255        if let Some(committee) = self
256            .0
257            .chain
258            .execution_state
259            .system
260            .committees
261            .get()
262            .get(&epoch)
263        {
264            certificate.check(committee)?;
265        } else {
266            let committees = self.0.storage.committees_for(epoch..=epoch).await?;
267            let Some(committee) = committees.get(&epoch) else {
268                let net_description = self
269                    .0
270                    .storage
271                    .read_network_description()
272                    .await?
273                    .ok_or_else(|| WorkerError::MissingNetworkDescription)?;
274                return Err(WorkerError::EventsNotFound(vec![EventId {
275                    chain_id: net_description.admin_chain_id,
276                    stream_id: StreamId::system(EPOCH_STREAM_NAME),
277                    index: epoch.0,
278                }]));
279            };
280            // This line is duplicated, but this avoids cloning and a lifetimes error.
281            certificate.check(committee)?;
282        }
283
284        // Certificate check passed - which means the blobs the block requires are legitimate and
285        // we can take note of it, so that if any are missing, we will accept them when the client
286        // sends them.
287        let required_blob_ids = block.required_blob_ids();
288        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
289        let blobs_result = self
290            .0
291            .get_required_blobs(required_blob_ids.iter().copied(), &created_blobs)
292            .await
293            .map(|blobs| blobs.into_values().collect::<Vec<_>>());
294
295        if let Ok(blobs) = &blobs_result {
296            self.0
297                .storage
298                .write_blobs_and_certificate(blobs, &certificate)
299                .await?;
300            let events = block
301                .body
302                .events
303                .iter()
304                .flatten()
305                .map(|event| (event.id(chain_id), event.value.clone()));
306            self.0.storage.write_events(events).await?;
307        }
308
309        // Update the blob state with last used certificate hash.
310        let blob_state = certificate.value().to_blob_state(blobs_result.is_ok());
311        let blob_ids = required_blob_ids.into_iter().collect::<Vec<_>>();
312        self.0
313            .storage
314            .maybe_write_blob_states(&blob_ids, blob_state)
315            .await?;
316
317        let mut blobs = blobs_result?
318            .into_iter()
319            .map(|blob| (blob.id(), blob))
320            .collect::<BTreeMap<_, _>>();
321
322        // If this block is higher than the next expected block in this chain, we're going
323        // to have a gap: do not execute this block, only update the outboxes and return.
324        if tip.next_block_height < height {
325            // Update the outboxes.
326            self.0.chain.preprocess_block(certificate.value()).await?;
327            // Persist chain.
328            self.save().await?;
329            let actions = self.0.create_network_actions().await?;
330            trace!("Preprocessed confirmed block {height} on chain {chain_id:.8}");
331            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
332                .await;
333            return Ok((self.chain_info_response(), actions));
334        }
335
336        // This should always be true for valid certificates.
337        ensure!(
338            tip.block_hash == block.header.previous_block_hash,
339            WorkerError::InvalidBlockChaining
340        );
341
342        // If we got here, `height` is equal to `tip.next_block_height` and the block is
343        // properly chained. Verify that the chain is active and that the epoch we used for
344        // verifying the certificate is actually the active one on the chain.
345        self.ensure_is_active().await?;
346        let (epoch, _) = self.0.chain.current_committee()?;
347        check_block_epoch(epoch, chain_id, block.header.epoch)?;
348
349        let published_blobs = block
350            .published_blob_ids()
351            .iter()
352            .filter_map(|blob_id| blobs.remove(blob_id))
353            .collect::<Vec<_>>();
354
355        // If height is zero, we haven't initialized the chain state or verified the epoch before -
356        // do it now.
357        // This will fail if the chain description blob is still missing - but that's alright,
358        // because we already wrote the blob state above, so the client can now upload the
359        // blob, which will get accepted, and retry.
360        if height == BlockHeight::ZERO {
361            self.ensure_is_active().await?;
362            let (epoch, _) = self.0.chain.current_committee()?;
363            check_block_epoch(epoch, chain_id, block.header.epoch)?;
364        }
365
366        // Execute the block and update inboxes.
367        let local_time = self.0.storage.clock().current_time();
368        let chain = &mut self.0.chain;
369        chain
370            .remove_bundles_from_inboxes(block.header.timestamp, block.body.incoming_bundles())
371            .await?;
372        let oracle_responses = Some(block.body.oracle_responses.clone());
373        let (proposed_block, outcome) = block.clone().into_proposal();
374        let verified_outcome = if let Some(mut execution_state) =
375            self.0.execution_state_cache.remove(&outcome.state_hash)
376        {
377            chain.execution_state = execution_state
378                .with_context(|ctx| {
379                    chain
380                        .execution_state
381                        .context()
382                        .clone_with_base_key(ctx.base_key().bytes.clone())
383                })
384                .await;
385            outcome.clone()
386        } else {
387            chain
388                .execute_block(
389                    &proposed_block,
390                    local_time,
391                    None,
392                    &published_blobs,
393                    oracle_responses,
394                )
395                .await?
396        };
397        // We should always agree on the messages and state hash.
398        ensure!(
399            outcome == verified_outcome,
400            WorkerError::IncorrectOutcome {
401                submitted: Box::new(outcome),
402                computed: Box::new(verified_outcome),
403            }
404        );
405        // Update the rest of the chain state.
406        chain
407            .apply_confirmed_block(certificate.value(), local_time)
408            .await?;
409        self.0.track_newly_created_chains(&proposed_block, &outcome);
410        let mut actions = self.0.create_network_actions().await?;
411        trace!("Processed confirmed block {height} on chain {chain_id:.8}");
412        let hash = certificate.hash();
413        let event_streams = certificate
414            .value()
415            .block()
416            .body
417            .events
418            .iter()
419            .flatten()
420            .map(|event| event.stream_id.clone())
421            .collect();
422        actions.notifications.push(Notification {
423            chain_id,
424            reason: Reason::NewBlock {
425                height,
426                hash,
427                event_streams,
428            },
429        });
430        // Persist chain.
431        self.save().await?;
432
433        self.0
434            .block_values
435            .insert(Cow::Owned(certificate.into_inner().into_inner()));
436
437        self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
438            .await;
439
440        Ok((self.chain_info_response(), actions))
441    }

443    /// Schedules a notification for when cross-chain messages are delivered up to the given
444    /// `height`.
445    #[instrument(level = "trace", skip(self, notify_when_messages_are_delivered))]
446    async fn register_delivery_notifier(
447        &mut self,
448        height: BlockHeight,
449        actions: &NetworkActions,
450        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
451    ) {
452        if let Some(notifier) = notify_when_messages_are_delivered {
453            if actions
454                .cross_chain_requests
455                .iter()
456                .any(|request| request.has_messages_lower_or_equal_than(height))
457            {
458                self.0.delivery_notifier.register(height, notifier);
459            } else {
460                // No need to wait. Also, cross-chain requests may not trigger the
461                // notifier later, even if we register it.
462                if let Err(()) = notifier.send(()) {
463                    warn!("Failed to notify message delivery to caller");
464                }
465            }
466        }
467    }

469    /// Updates the chain's inboxes, receiving messages from a cross-chain update.
470    #[instrument(level = "trace", skip(self, bundles))]
471    pub(super) async fn process_cross_chain_update(
472        &mut self,
473        origin: ChainId,
474        bundles: Vec<(Epoch, MessageBundle)>,
475    ) -> Result<Option<BlockHeight>, WorkerError> {
476        // Only process certificates with relevant heights and epochs.
477        let next_height_to_receive = self.0.chain.next_block_height_to_receive(&origin).await?;
478        let last_anticipated_block_height =
479            self.0.chain.last_anticipated_block_height(&origin).await?;
480        let helper = CrossChainUpdateHelper::new(&self.0.config, &self.0.chain);
481        let recipient = self.0.chain_id();
482        let bundles = helper.select_message_bundles(
483            &origin,
484            recipient,
485            next_height_to_receive,
486            last_anticipated_block_height,
487            bundles,
488        )?;
489        let Some(last_updated_height) = bundles.last().map(|bundle| bundle.height) else {
490            return Ok(None);
491        };
492        // Process the received messages in certificates.
493        let local_time = self.0.storage.clock().current_time();
494        let mut previous_height = None;
495        for bundle in bundles {
496            let add_to_received_log = previous_height != Some(bundle.height);
497            previous_height = Some(bundle.height);
498            // Update the staged chain state with the received block.
499            self.0
500                .chain
501                .receive_message_bundle(&origin, bundle, local_time, add_to_received_log)
502                .await?;
503        }
504        if !self.0.config.allow_inactive_chains && !self.0.chain.is_active() {
505            // Refuse to create a chain state if the chain is still inactive by
506            // now. Accordingly, do not send a confirmation, so that the
507            // cross-chain update is retried later.
508            warn!(
509                "Refusing to deliver messages to {recipient:?} from {origin:?} \
510                at height {last_updated_height} because the recipient is still inactive",
511            );
512            return Ok(None);
513        }
514        // Save the chain.
515        self.save().await?;
516        Ok(Some(last_updated_height))
517    }

519    /// Handles the cross-chain request confirming that the recipient was updated.
520    pub(super) async fn confirm_updated_recipient(
521        &mut self,
522        recipient: ChainId,
523        latest_height: BlockHeight,
524    ) -> Result<(), WorkerError> {
525        let fully_delivered = self
526            .0
527            .chain
528            .mark_messages_as_received(&recipient, latest_height)
529            .await?
530            && self
531                .0
532                .all_messages_to_tracked_chains_delivered_up_to(latest_height)
533                .await?;
534
535        self.save().await?;
536
537        if fully_delivered {
538            self.0.delivery_notifier.notify(latest_height);
539        }
540
541        Ok(())
542    }

555    /// Attempts to vote for a leader timeout, if possible.
556    pub(super) async fn vote_for_leader_timeout(
557        &mut self,
558        height: BlockHeight,
559        round: Round,
560    ) -> Result<(), WorkerError> {
561        let chain = &mut self.0.chain;
562        ensure!(
563            height == chain.tip_state.get().next_block_height,
564            WorkerError::UnexpectedBlockHeight {
565                expected_block_height: chain.tip_state.get().next_block_height,
566                found_block_height: height
567            }
568        );
569        let epoch = chain.execution_state.system.epoch.get();
570        let chain_id = chain.chain_id();
571        let key_pair = self.0.config.key_pair();
572        let local_time = self.0.storage.clock().current_time();
573        if chain
574            .manager
575            .create_timeout_vote(chain_id, height, round, *epoch, key_pair, local_time)?
576        {
577            self.save().await?;
578        }
579        Ok(())
580    }

582    /// Votes for falling back to a public chain.
583    pub(super) async fn vote_for_fallback(&mut self) -> Result<(), WorkerError> {
584        let chain = &mut self.0.chain;
585        if let (epoch, Some(entry)) = (
586            chain.execution_state.system.epoch.get(),
587            chain.unskippable_bundles.front(),
588        ) {
589            let elapsed = self
590                .0
591                .storage
592                .clock()
593                .current_time()
594                .delta_since(entry.seen);
595            if elapsed >= chain.ownership().timeout_config.fallback_duration {
596                let chain_id = chain.chain_id();
597                let height = chain.tip_state.get().next_block_height;
598                let key_pair = self.0.config.key_pair();
599                if chain
600                    .manager
601                    .vote_fallback(chain_id, height, *epoch, key_pair)
602                {
603                    self.save().await?;
604                }
605            }
606        }
607        Ok(())
608    }

610    pub(super) async fn handle_pending_blob(
611        &mut self,
612        blob: Blob,
613    ) -> Result<ChainInfoResponse, WorkerError> {
614        let mut was_expected = self
615            .0
616            .chain
617            .pending_validated_blobs
618            .maybe_insert(&blob)
619            .await?;
620        for (_, mut pending_blobs) in self
621            .0
622            .chain
623            .pending_proposed_blobs
624            .try_load_all_entries_mut()
625            .await?
626        {
627            if !pending_blobs.validated.get() {
628                let (_, committee) = self.0.chain.current_committee()?;
629                let policy = committee.policy();
630                policy
631                    .check_blob_size(blob.content())
632                    .with_execution_context(ChainExecutionContext::Block)?;
633                ensure!(
634                    u64::try_from(pending_blobs.pending_blobs.count().await?)
635                        .is_ok_and(|count| count < policy.maximum_published_blobs),
636                    WorkerError::TooManyPublishedBlobs(policy.maximum_published_blobs)
637                );
638            }
639            was_expected = was_expected || pending_blobs.maybe_insert(&blob).await?;
640        }
641        ensure!(was_expected, WorkerError::UnexpectedBlob);
642        self.save().await?;
643        Ok(self.chain_info_response())
644    }

646    /// Returns a stored [`Certificate`] for the chain's block at the requested [`BlockHeight`].
647    #[cfg(with_testing)]
648    pub(super) async fn read_certificate(
649        &mut self,
650        height: BlockHeight,
651    ) -> Result<Option<ConfirmedBlockCertificate>, WorkerError> {
652        self.ensure_is_active().await?;
653        let certificate_hash = match self.0.chain.confirmed_log.get(height.try_into()?).await? {
654            Some(hash) => hash,
655            None => return Ok(None),
656        };

661            .await?
662            .ok_or_else(|| WorkerError::ReadCertificatesError(vec![certificate_hash]))?;
663        Ok(Some(certificate))
664    }

724    /// Validates and executes a block proposed to extend this chain.
725    pub(super) async fn handle_block_proposal(
726        &mut self,
727        proposal: BlockProposal,
728    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
729        self.ensure_is_active().await?;
730        proposal
731            .check_invariants()
732            .map_err(|msg| WorkerError::InvalidBlockProposal(msg.to_string()))?;
733        proposal.check_signature()?;
734        let owner = proposal.owner();
735        let BlockProposal {
736            content,
737            original_proposal,
738            signature: _,
739        } = &proposal;
740        let block = &content.block;
741        let chain = &self.0.chain;
742        // Check the epoch.
743        let (epoch, committee) = chain.current_committee()?;
744        super::check_block_epoch(epoch, block.chain_id, block.epoch)?;
745        let policy = committee.policy().clone();
746        block.check_proposal_size(policy.maximum_block_proposal_size)?;
747        // Check the authentication of the block.
748        ensure!(
749            chain.manager.verify_owner(&owner, proposal.content.round)?,
750            WorkerError::InvalidOwner
751        );
752        match original_proposal {
753            None => {
754                if let Some(signer) = block.authenticated_signer {
755                    // Check the authentication of the operations in the new block.
756                    ensure!(signer == owner, WorkerError::InvalidSigner(owner));
757                }
758            }
759            Some(OriginalProposal::Regular { certificate }) => {
760                // Verify that this block has been validated by a quorum before.
761                certificate.check(committee)?;
762            }
763            Some(OriginalProposal::Fast(signature)) => {
764                let original_proposal = BlockProposal {
765                    content: ProposalContent {
766                        block: content.block.clone(),
767                        round: Round::Fast,
768                        outcome: None,
769                    },
770                    signature: *signature,
771                    original_proposal: None,
772                };
773                let super_owner = original_proposal.owner();
774                ensure!(
775                    chain
776                        .manager
777                        .ownership
778                        .get()
779                        .super_owners
780                        .contains(&super_owner),
781                    WorkerError::InvalidOwner
782                );
783                if let Some(signer) = block.authenticated_signer {
784                    // Check the authentication of the operations in the new block.
785                    ensure!(signer == super_owner, WorkerError::InvalidSigner(signer));
786                }
787                original_proposal.check_signature()?;
788            }
789        }
790        // Check if the chain is ready for this new block proposal.
791        chain.tip_state.get().verify_block_chaining(block)?;
792        if chain.manager.check_proposed_block(&proposal)? == manager::Outcome::Skip {
793            // We already voted for this block.
794            return Ok((self.chain_info_response(), NetworkActions::default()));
795        }
796
797        let published_blobs = self.load_proposal_blobs(&proposal).await?;
798        let ProposalContent {
799            block,
800            round,
801            outcome,
802        } = content;
803
804        let local_time = self.0.storage.clock().current_time();
805        ensure!(
806            block.timestamp.duration_since(local_time) <= self.0.config.grace_period,
807            WorkerError::InvalidTimestamp
808        );
809        self.0.storage.clock().sleep_until(block.timestamp).await;
810        let local_time = self.0.storage.clock().current_time();
811
812        self.0
813            .chain
814            .remove_bundles_from_inboxes(block.timestamp, block.incoming_bundles())
815            .await?;
816        let outcome = if let Some(outcome) = outcome {
817            outcome.clone()
818        } else {
819            self.execute_block(block, local_time, round.multi_leader(), &published_blobs)
820                .await?
821        };
822
823        ensure!(
824            !round.is_fast() || !outcome.has_oracle_responses(),
825            WorkerError::FastBlockUsingOracles
826        );
827        let chain = &mut self.0.chain;
828        // Check if the counters of tip_state would be valid.
829        chain
830            .tip_state
831            .get_mut()
832            .update_counters(&block.transactions, &outcome.messages)?;
833        // Verify that the resulting chain would have no unconfirmed incoming messages.
834        chain.validate_incoming_bundles().await?;
835        // Don't save the changes since the block is not confirmed yet.
836        chain.rollback();
837
838        // Create the vote and store it in the chain state.
839        let block = outcome.with(proposal.content.block.clone());
840        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
841        let blobs = self
842            .0
843            .get_required_blobs(proposal.expected_blob_ids(), &created_blobs)
844            .await?;
845        let key_pair = self.0.config.key_pair();
846        let manager = &mut self.0.chain.manager;
847        match manager.create_vote(proposal, block, key_pair, local_time, blobs)? {
848            // Cache the value we voted on, so the client doesn't have to send it again.
849            Some(Either::Left(vote)) => {
850                self.0
851                    .block_values
852                    .insert(Cow::Borrowed(vote.value.inner()));
853            }
854            Some(Either::Right(vote)) => {
855                self.0
856                    .block_values
857                    .insert(Cow::Borrowed(vote.value.inner()));
858            }
859            None => (),
860        }
861        self.save().await?;
862        let actions = self.0.create_network_actions().await?;
863        Ok((self.chain_info_response(), actions))
864    }

866    /// Prepares a [`ChainInfoResponse`] for a [`ChainInfoQuery`].
867    pub(super) async fn prepare_chain_info_response(
868        &mut self,
869        query: ChainInfoQuery,
870    ) -> Result<ChainInfoResponse, WorkerError> {
871        self.ensure_is_active().await?;
872        let chain = &self.0.chain;
873        let mut info = ChainInfo::from(chain);
874        if query.request_committees {
875            info.requested_committees = Some(chain.execution_state.system.committees.get().clone());
876        }
877        if query.request_owner_balance == AccountOwner::CHAIN {
878            info.requested_owner_balance = Some(*chain.execution_state.system.balance.get());
879        } else {
880            info.requested_owner_balance = chain
881                .execution_state
882                .system
883                .balances
884                .get(&query.request_owner_balance)
885                .await?;
886        }
887        if let Some(next_block_height) = query.test_next_block_height {
888            ensure!(
889                chain.tip_state.get().next_block_height == next_block_height,
890                WorkerError::UnexpectedBlockHeight {
891                    expected_block_height: next_block_height,
892                    found_block_height: chain.tip_state.get().next_block_height
893                }
894            );
895        }
896        if query.request_pending_message_bundles {
897            let mut messages = Vec::new();
898            let pairs = chain.inboxes.try_load_all_entries().await?;
899            let action = if *chain.execution_state.system.closed.get() {
900                MessageAction::Reject
901            } else {
902                MessageAction::Accept
903            };
904            for (origin, inbox) in pairs {
905                for bundle in inbox.added_bundles.elements().await? {
906                    messages.push(IncomingBundle {
907                        origin,
908                        bundle,
909                        action,
910                    });
911                }
912            }
913
914            info.requested_pending_message_bundles = messages;
915        }
916        if let Some(range) = query.request_sent_certificate_hashes_in_range {
917            let start: usize = range.start.try_into()?;
918            let end = match range.limit {
919                None => chain.confirmed_log.count(),
920                Some(limit) => start
921                    .checked_add(usize::try_from(limit).map_err(|_| ArithmeticError::Overflow)?)
922                    .ok_or(ArithmeticError::Overflow)?
923                    .min(chain.confirmed_log.count()),
924            };
925            let keys = chain.confirmed_log.read(start..end).await?;
926            info.requested_sent_certificate_hashes = keys;
927        }
928        if let Some(start) = query.request_received_log_excluding_first_n {
929            let start = usize::try_from(start).map_err(|_| ArithmeticError::Overflow)?;
930            info.requested_received_log = chain.received_log.read(start..).await?;
931        }
932        if query.request_manager_values {
933            info.manager.add_values(&chain.manager);
934        }
935        Ok(ChainInfoResponse::new(info, self.0.config.key_pair()))
936    }

941        block: &ProposedBlock,

998/// Helper type for handling cross-chain updates.
999pub(crate) struct CrossChainUpdateHelper<'a> {
1000    pub allow_messages_from_deprecated_epochs: bool,
1001    pub current_epoch: Epoch,
1002    pub committees: &'a BTreeMap<Epoch, Committee>,
1003}
1004
1005impl<'a> CrossChainUpdateHelper<'a> {
1006    /// Creates a new [`CrossChainUpdateHelper`].
1007    pub fn new<C>(config: &ChainWorkerConfig, chain: &'a ChainStateView<C>) -> Self
1008    where
1009        C: Context + Clone + Send + Sync + 'static,
1010    {
1011        CrossChainUpdateHelper {
1012            allow_messages_from_deprecated_epochs: config.allow_messages_from_deprecated_epochs,
1013            current_epoch: *chain.execution_state.system.epoch.get(),
1014            committees: chain.execution_state.system.committees.get(),
1015        }
1016    }
1017
1018    /// Checks basic invariants and deals with repeated heights and deprecated epochs.
1019    /// * Returns a range of message bundles that are both new to us and not relying on
1020    ///   an untrusted set of validators.
1021    /// * In the case of validators, if the epoch(s) of the highest bundles are not
1022    ///   trusted, we only accept bundles that contain messages that were already
1023    ///   executed by anticipation (i.e. received in certified blocks).
1024    /// * Basic invariants are checked for good measure. We still crucially trust
1025    ///   the worker of the sending chain to have verified and executed the blocks
1026    ///   correctly.
1027    pub fn select_message_bundles(
1028        &self,
1029        origin: &'a ChainId,
1030        recipient: ChainId,
1031        next_height_to_receive: BlockHeight,
1032        last_anticipated_block_height: Option<BlockHeight>,
1033        mut bundles: Vec<(Epoch, MessageBundle)>,
1034    ) -> Result<Vec<MessageBundle>, WorkerError> {
1035        let mut latest_height = None;
1036        let mut skipped_len = 0;
1037        let mut trusted_len = 0;
1038        for (i, (epoch, bundle)) in bundles.iter().enumerate() {
1039            // Make sure that heights are not decreasing.
1040            ensure!(
1041                latest_height <= Some(bundle.height),
1042                WorkerError::InvalidCrossChainRequest
1043            );
1044            latest_height = Some(bundle.height);
1045            // Check if the block has been received already.
1046            if bundle.height < next_height_to_receive {
1047                skipped_len = i + 1;
1048            }
1049            // Check if the height is trusted or the epoch is trusted.
1050            if self.allow_messages_from_deprecated_epochs
1051                || Some(bundle.height) <= last_anticipated_block_height
1052                || *epoch >= self.current_epoch
1053                || self.committees.contains_key(epoch)
1054            {
1055                trusted_len = i + 1;
1056            }
1057        }
1058        if skipped_len > 0 {
1059            let (_, sample_bundle) = &bundles[skipped_len - 1];
1060            debug!(
1061                "Ignoring repeated messages to {recipient:.8} from {origin:} at height {}",
1062                sample_bundle.height,
1063            );
1064        }
1065        if skipped_len < bundles.len() && trusted_len < bundles.len() {
1066            let (sample_epoch, sample_bundle) = &bundles[trusted_len];
1067            warn!(
1068                "Refusing messages to {recipient:.8} from {origin:} at height {} \
1069                 because the epoch {} is not trusted any more",
1070                sample_bundle.height, sample_epoch,
1071            );
1072        }
1073        let bundles = if skipped_len < trusted_len {
1074            bundles
1075                .drain(skipped_len..trusted_len)
1076                .map(|(_, bundle)| bundle)
1077                .collect()
1078        } else {
1079            vec![]
1080        };
1081        Ok(bundles)
1082    }
