39/// The default amount of time we wait for additional validators to contribute
40/// to the result, as a fraction of how long it took to reach a quorum.
41pub const DEFAULT_GRACE_PERIOD: f64 = 0.2;
42/// The maximum timeout for requests to a stake-weighted quorum if no quorum is reached.
43const MAX_TIMEOUT: Duration = Duration::from_secs(60 * 60 * 24); // 1 day.
44
45/// Used for `communicate_chain_action`
46#[derive(Clone)]
47pub enum CommunicateAction {
48    SubmitBlock {
49        proposal: Box<BlockProposal>,
50        blob_ids: Vec<BlobId>,
51    },
52    FinalizeBlock {
53        certificate: Box<ValidatedBlockCertificate>,
54        delivery: CrossChainMessageDelivery,
55    },
56    RequestTimeout {
57        chain_id: ChainId,
58        height: BlockHeight,
59        round: Round,
60    },
61}
62
63impl CommunicateAction {
64    /// The round to which this action pertains.
65    pub fn round(&self) -> Round {
66        match self {
67            CommunicateAction::SubmitBlock { proposal, .. } => proposal.content.round,
68            CommunicateAction::FinalizeBlock { certificate, .. } => certificate.round,
69            CommunicateAction::RequestTimeout { round, .. } => *round,
70        }
71    }
72}

83/// An error result for requests to a stake-weighted quorum.
84#[derive(Error, Debug)]
85pub enum CommunicationError<E: fmt::Debug> {
86    /// No consensus is possible since validators returned different possibilities
87    /// for the next block
88    #[error(
89        "No error but failed to find a consensus block. Consensus threshold: {0}, Proposals: {1:?}"
90    )]
91    NoConsensus(u64, Vec<(u64, usize)>),
92    /// A single error that was returned by a sufficient number of nodes to be trusted as
93    /// valid.
94    #[error("Failed to communicate with a quorum of validators: {0}")]
95    Trusted(E),
96    /// No single error reached the validity threshold so we're returning a sample of
97    /// errors for debugging purposes, together with their weight.
98    #[error("Failed to communicate with a quorum of validators:\n{:#?}", .0)]
99    Sample(Vec<(E, u64)>),
100}
101
102/// Executes a sequence of actions in parallel for all validators.
103///
104/// Tries to stop early when a quorum is reached. If `grace_period` is specified, other validators
105/// are given additional time to contribute to the result. The grace period is calculated as a fraction
106/// (defaulting to `DEFAULT_GRACE_PERIOD`) of the time taken to reach quorum.
107pub async fn communicate_with_quorum<'a, A, V, K, F, R, G>(
108    validator_clients: &'a [RemoteNode<A>],
109    committee: &Committee,
110    group_by: G,
111    execute: F,
112    // Grace period as a fraction of time taken to reach quorum
113    grace_period: f64,
114) -> Result<(K, Vec<(ValidatorPublicKey, V)>), CommunicationError<NodeError>>
115where
116    A: ValidatorNode + Clone + 'static,
117    F: Clone + Fn(RemoteNode<A>) -> R,
118    R: Future<Output = Result<V, ChainClientError>> + 'a,
119    G: Fn(&V) -> K,
120    K: Hash + PartialEq + Eq + Clone + 'static,
121    V: 'static,
122{
123    let mut responses: futures::stream::FuturesUnordered<_> = validator_clients
124        .iter()
125        .filter_map(|remote_node| {
126            if committee.weight(&remote_node.public_key) == 0 {
127                // This should not happen but better prevent it because certificates
128                // are not allowed to include votes with weight 0.
129                return None;
130            }
131            let execute = execute.clone();
132            let remote_node = remote_node.clone();
133            Some(async move { (remote_node.public_key, execute(remote_node).await) })
134        })
135        .collect();
136
137    let start_time = Instant::now();
138    let mut end_time: Option<Instant> = None;
139    let mut remaining_votes = committee.total_votes();
140    let mut highest_key_score = 0;
141    let mut value_scores: HashMap<K, (u64, Vec<(ValidatorPublicKey, V)>)> = HashMap::new();
142    let mut error_scores = HashMap::new();
143
144    'vote_wait: while let Ok(Some((name, result))) = timeout(
145        end_time.map_or(MAX_TIMEOUT, |t| t.saturating_duration_since(Instant::now())),
146        responses.next(),
147    )
148    .await
149    {
150        remaining_votes -= committee.weight(&name);
151        match result {
152            Ok(value) => {
153                let key = group_by(&value);
154                let entry = value_scores.entry(key.clone()).or_insert((0, Vec::new()));
155                entry.0 += committee.weight(&name);
156                entry.1.push((name, value));
157                highest_key_score = highest_key_score.max(entry.0);
158            }
159            Err(err) => {
160                // TODO(#2857): Handle non-remote errors properly.
161                let err = match err {
162                    ChainClientError::RemoteNodeError(err) => err,
163                    err => NodeError::ResponseHandlingError {
164                        error: err.to_string(),
165                    },
166                };
167                let entry = error_scores.entry(err.clone()).or_insert(0);
168                *entry += committee.weight(&name);
169                if *entry >= committee.validity_threshold() {
170                    // At least one honest node returned this error.
171                    // No quorum can be reached, so return early.
172                    return Err(CommunicationError::Trusted(err));
173                }
174            }
175        }
176        // If it becomes clear that no key can reach a quorum, break early.
177        if highest_key_score + remaining_votes < committee.quorum_threshold() {
178            break 'vote_wait;
179        }
180
181        // If a key reaches a quorum, wait for the grace period to collect more values
182        // or error information and then stop.
183        if end_time.is_none() && highest_key_score >= committee.quorum_threshold() {
184            end_time = Some(Instant::now() + start_time.elapsed().mul_f64(grace_period));
185        }
186    }
187
188    let scores = value_scores
189        .values()
190        .map(|(weight, values)| (*weight, values.len()))
191        .collect();
192    // If a key has a quorum, return it with its values.
193    if let Some((key, (_, values))) = value_scores
194        .into_iter()
195        .find(|(_, (score, _))| *score >= committee.quorum_threshold())
196    {
197        return Ok((key, values));
198    }
199
200    if error_scores.is_empty() {
201        return Err(CommunicationError::NoConsensus(
202            committee.quorum_threshold(),
203            scores,
204        ));
205    }
206
207    // No specific error is available to report reliably.
208    let mut sample = error_scores.into_iter().collect::<Vec<_>>();
209    sample.sort_by_key(|(_, score)| std::cmp::Reverse(*score));
210    sample.truncate(4);
211    Err(CommunicationError::Sample(sample))
212}

219    async fn send_confirmed_certificate(
220        &mut self,
221        certificate: GenericCertificate<ConfirmedBlock>,
222        delivery: CrossChainMessageDelivery,
223    ) -> Result<Box<ChainInfo>, ChainClientError> {
224        let result = self
225            .remote_node
226            .handle_optimized_confirmed_certificate(&certificate, delivery)
227            .await;
228
229        Ok(match &result {
230            Err(original_err @ NodeError::BlobsNotFound(blob_ids)) => {
231                self.remote_node
232                    .check_blobs_not_found(&certificate, blob_ids)?;
233                // The certificate is confirmed, so the blobs must be in storage.
234                let maybe_blobs = self.local_node.read_blobs_from_storage(blob_ids).await?;
235                let blobs = maybe_blobs.ok_or_else(|| original_err.clone())?;
236                self.remote_node.node.upload_blobs(blobs.clone()).await?;
237                self.remote_node
238                    .handle_confirmed_certificate(certificate, delivery)
239                    .await
240            }
241            _ => result,
242        }?)
243    }
244
245    async fn send_validated_certificate(
246        &mut self,
247        certificate: GenericCertificate<ValidatedBlock>,
248        delivery: CrossChainMessageDelivery,
249    ) -> Result<Box<ChainInfo>, ChainClientError> {
250        let result = self
251            .remote_node
252            .handle_optimized_validated_certificate(&certificate, delivery)
253            .await;
254
255        Ok(match &result {
256            Err(original_err @ NodeError::BlobsNotFound(blob_ids)) => {
257                self.remote_node
258                    .check_blobs_not_found(&certificate, blob_ids)?;
259                let chain_id = certificate.inner().chain_id();
260                // The certificate is for a validated block, i.e. for our locking block.
261                // Take the missing blobs from our local chain manager.
262                let blobs = self
263                    .local_node
264                    .get_locking_blobs(blob_ids, chain_id)
265                    .await?
266                    .ok_or_else(|| original_err.clone())?;
267                self.remote_node.send_pending_blobs(chain_id, blobs).await?;
268                self.remote_node
269                    .handle_validated_certificate(certificate)
270                    .await
271            }
272            _ => result,
273        }?)
274    }

276    async fn send_block_proposal(
277        &mut self,
278        proposal: Box<BlockProposal>,
279        mut blob_ids: Vec<BlobId>,
280    ) -> Result<Box<ChainInfo>, ChainClientError> {
281        let chain_id = proposal.content.block.chain_id;
282        let mut sent_cross_chain_updates = false;
283        let mut publisher_chain_ids_sent = BTreeSet::new();
284        loop {
285            match self
286                .remote_node
287                .handle_block_proposal(proposal.clone())
288                .await
289            {
290                Ok(info) => return Ok(info),
291                Err(NodeError::WrongRound(_round)) => {
292                    // The proposal is for a different round, so we need to update the validator.
293                    // TODO: this should probably be more specific as to which rounds are retried.
294                    self.send_chain_information(
295                        chain_id,
296                        proposal.content.block.height,
297                        CrossChainMessageDelivery::NonBlocking,
298                    )
299                    .await?;
300                }
301                Err(NodeError::UnexpectedBlockHeight {
302                    expected_block_height,
303                    found_block_height,
304                }) if expected_block_height < found_block_height => {
305                    // The proposal is for a later block height, so we need to update the validator.
306                    self.send_chain_information(
307                        chain_id,
308                        found_block_height,
309                        CrossChainMessageDelivery::NonBlocking,
310                    )
311                    .await?;
312                }
313                Err(NodeError::MissingCrossChainUpdate { .. }) if !sent_cross_chain_updates => {
314                    sent_cross_chain_updates = true;
315                    // Some received certificates may be missing for this validator
316                    // (e.g. to create the chain or make the balance sufficient) so we are going to
317                    // synchronize them now and retry.
318                    self.send_chain_information_for_senders(chain_id).await?;
319                }
320                Err(NodeError::EventsNotFound(event_ids)) => {
321                    let mut publisher_heights = BTreeMap::new();
322                    let new_chain_ids = event_ids
323                        .iter()
324                        .map(|event_id| event_id.chain_id)
325                        .filter(|chain_id| !publisher_chain_ids_sent.contains(chain_id))
326                        .collect::<BTreeSet<_>>();
327                    ensure!(
328                        !new_chain_ids.is_empty(),
329                        NodeError::EventsNotFound(event_ids)
330                    );
331                    for chain_id in new_chain_ids {
332                        let info = self.local_node.chain_info(chain_id).await?;
333                        publisher_heights.insert(chain_id, info.next_block_height);
334                        publisher_chain_ids_sent.insert(chain_id);
335                    }
336                    self.send_chain_info_up_to_heights(
337                        publisher_heights,
338                        CrossChainMessageDelivery::NonBlocking,
339                    )
340                    .await?;
341                }
342                Err(NodeError::BlobsNotFound(_) | NodeError::InactiveChain(_))
343                    if !blob_ids.is_empty() =>
344                {
345                    // For `BlobsNotFound`, we assume that the local node should already be
346                    // updated with the needed blobs, so sending the chain information about the
347                    // certificates that last used the blobs to the validator node should be enough.
348                    let published_blob_ids =
349                        BTreeSet::from_iter(proposal.content.block.published_blob_ids());
350                    blob_ids.retain(|blob_id| !published_blob_ids.contains(blob_id));
351                    let mut published_blobs = Vec::new();
352                    {
353                        let chain = self.local_node.chain_state_view(chain_id).await?;
354                        for blob_id in published_blob_ids {
355                            published_blobs
356                                .extend(chain.manager.proposed_blobs.get(&blob_id).await?);
357                        }
358                    }
359                    self.remote_node
360                        .send_pending_blobs(chain_id, published_blobs)
361                        .await?;
362                    let missing_blob_ids = self
363                        .remote_node
364                        .node
365                        .missing_blob_ids(mem::take(&mut blob_ids))
366                        .await?;
367                    let blob_states = self
368                        .local_node
369                        .read_blob_states_from_storage(&missing_blob_ids)
370                        .await?;
371                    let mut chain_heights = BTreeMap::new();
372                    for blob_state in blob_states {
373                        let block_chain_id = blob_state.chain_id;
374                        let block_height = blob_state.block_height.try_add_one()?;
375                        chain_heights
376                            .entry(block_chain_id)
377                            .and_modify(|h| *h = block_height.max(*h))
378                            .or_insert(block_height);
379                    }
380
381                    self.send_chain_info_up_to_heights(
382                        chain_heights,
383                        CrossChainMessageDelivery::NonBlocking,
384                    )
385                    .await?;
386                }
387                // Fail immediately on other errors.
388                Err(err) => return Err(err.into()),
389            }
390        }
391    }

393    pub async fn send_chain_information(
394        &mut self,
395        chain_id: ChainId,
396        target_block_height: BlockHeight,
397        delivery: CrossChainMessageDelivery,
398    ) -> Result<(), ChainClientError> {
399        // Figure out which certificates this validator is missing. In many cases, it's just the
400        // last one, so we optimistically send that one right away.
401        let remote_info = if let Ok(height) = target_block_height.try_sub_one() {
402            let chain = self.local_node.chain_state_view(chain_id).await?;
403            if let Some(hash) = chain
404                .block_hashes(height..target_block_height)
405                .await?
406                .first()
407            {
408                let certificate = self
409                    .local_node
410                    .storage_client()
411                    .read_certificate(*hash)
412                    .await?
413                    .ok_or_else(|| ChainClientError::MissingConfirmedBlock(*hash))?;
414                match self.send_confirmed_certificate(certificate, delivery).await {
415                    Err(ChainClientError::RemoteNodeError(NodeError::EventsNotFound(
416                        event_ids,
417                    ))) if event_ids.iter().all(|event_id| {
418                        event_id.stream_id.application_id == GenericApplicationId::System
419                    }) =>
420                    {
421                        // The chain is missing epoch events. Send all blocks.
422                        let query = ChainInfoQuery::new(chain_id);
423                        self.remote_node.handle_chain_info_query(query).await?
424                    }
425                    Err(err) => return Err(err),
426                    Ok(info) => info,
427                }
428            } else {
429                // We don't have the block at the specified height. Send all blocks.
430                tracing::warn!(
431                    "send_chain_information called with height {target_block_height},
432                    but {chain_id:.8} does not have that block"
433                );
434                let query = ChainInfoQuery::new(chain_id);
435                self.remote_node.handle_chain_info_query(query).await?
436            }
437        } else {
438            let query = ChainInfoQuery::new(chain_id);
439            self.remote_node.handle_chain_info_query(query).await?
440        };
441        let initial_block_height = remote_info.next_block_height;
442        // Obtain the missing blocks and the manager state from the local node.
443        let range = initial_block_height..target_block_height;
444        let keys = {
445            let chain = self.local_node.chain_state_view(chain_id).await?;
446            chain.block_hashes(range).await?
447        };
448        if !keys.is_empty() {
449            // Send the requested certificates in order.
450            let storage = self.local_node.storage_client();
451            let certificates = storage.read_certificates(keys.clone()).await?;
452            let certificates = match ResultReadCertificates::new(certificates, keys) {
453                ResultReadCertificates::Certificates(certificates) => certificates,
454                ResultReadCertificates::InvalidHashes(hashes) => {
455                    return Err(ChainClientError::ReadCertificatesError(hashes))
456                }
457            };
458            for certificate in certificates {
459                self.send_confirmed_certificate(certificate, delivery)
460                    .await?;
461            }
462        }
463        // If the remote node is missing a timeout certificate, send it as well.
464        let local_info = self.local_node.chain_info(chain_id).await?;
465        if let Some(cert) = local_info.manager.timeout {
466            if (local_info.next_block_height, cert.round)
467                >= (
468                    remote_info.next_block_height,
469                    remote_info.manager.current_round,
470                )
471            {
472                self.remote_node.handle_timeout_certificate(*cert).await?;
473            }
474        }
475        Ok(())
476    }

478    async fn send_chain_info_up_to_heights(
479        &mut self,
480        chain_heights: BTreeMap<ChainId, BlockHeight>,
481        delivery: CrossChainMessageDelivery,
482    ) -> Result<(), ChainClientError> {
483        let stream =
484            FuturesUnordered::from_iter(chain_heights.into_iter().map(|(chain_id, height)| {
485                let mut updater = self.clone();
486                async move {
487                    updater
488                        .send_chain_information(chain_id, height, delivery)
489                        .await
490                }
491            }));
492        stream.try_collect::<Vec<_>>().await?;
493        Ok(())
494    }
495
496    async fn send_chain_information_for_senders(
497        &mut self,
498        chain_id: ChainId,
499    ) -> Result<(), ChainClientError> {
500        let mut sender_heights = BTreeMap::new();
501        {
502            let chain = self.local_node.chain_state_view(chain_id).await?;
503            let pairs = chain.inboxes.try_load_all_entries().await?;
504            for (origin, inbox) in pairs {
505                let inbox_next_height = inbox.next_block_height_to_receive()?;
506                sender_heights
507                    .entry(origin)
508                    .and_modify(|h| *h = inbox_next_height.max(*h))
509                    .or_insert(inbox_next_height);
510            }
511        }
512
513        self.send_chain_info_up_to_heights(sender_heights, CrossChainMessageDelivery::Blocking)
514            .await?;
515        Ok(())
516    }

518    pub async fn send_chain_update(
519        &mut self,
520        action: CommunicateAction,
521    ) -> Result<LiteVote, ChainClientError> {
522        let chain_id = match &action {
523            CommunicateAction::SubmitBlock { proposal, .. } => proposal.content.block.chain_id,
524            CommunicateAction::FinalizeBlock { certificate, .. } => {
525                certificate.inner().block().header.chain_id
526            }
527            CommunicateAction::RequestTimeout { chain_id, .. } => *chain_id,
528        };
529        // Send the block proposal, certificate or timeout request and return a vote.
530        let vote = match action {
531            CommunicateAction::SubmitBlock { proposal, blob_ids } => {
532                let info = self.send_block_proposal(proposal, blob_ids).await?;
533                info.manager.pending.ok_or_else(|| {
534                    NodeError::MissingVoteInValidatorResponse("submit a block proposal".into())
535                })?
536            }
537            CommunicateAction::FinalizeBlock {
538                certificate,
539                delivery,
540            } => {
541                let info = self
542                    .send_validated_certificate(*certificate, delivery)
543                    .await?;
544                info.manager.pending.ok_or_else(|| {
545                    NodeError::MissingVoteInValidatorResponse("finalize a block".into())
546                })?
547            }
548            CommunicateAction::RequestTimeout { round, height, .. } => {
549                let query = ChainInfoQuery::new(chain_id).with_timeout(height, round);
550                let info = self.remote_node.handle_chain_info_query(query).await?;
551                info.manager.timeout_vote.ok_or_else(|| {
552                    NodeError::MissingVoteInValidatorResponse("request a timeout".into())
553                })?
554            }
555        };
556        vote.check(self.remote_node.public_key)?;
557        Ok(vote)
558    }
